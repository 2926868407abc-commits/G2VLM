# Tested smoke-train stack: Python 3.10, torch 2.5.1+cu121, CUDA 12.x, flash-attn 2.7.4.post1.
#
# Install order matters on a fresh server:
# 1. Create and activate an isolated env first. If conda is not available, use python3 -m venv.
# 2. Install torch before this file:
#      python -m pip install torch==2.5.1 torchvision==0.20.1 torchaudio==2.5.1 --index-url https://download.pytorch.org/whl/cu121
# 3. Install this file:
#      python -m pip install -r requirements.txt
# 4. Install flash-attn after torch, without dependencies, so pip does not upgrade torch:
#      python -m pip install --no-deps https://github.com/Dao-AILab/flash-attention/releases/download/v2.7.4.post1/flash_attn-2.7.4.post1+cu12torch2.5cxx11abiFALSE-cp310-cp310-linux_x86_64.whl
#
# Do not upgrade huggingface_hub to 1.x inside this training env. transformers==4.49.0
# requires huggingface_hub<1.0, and the newer hub CLI can break this stack. Use a
# separate downloader env if you need a newer `hf` command for dataset download.

#      export HF_ENDPOINT=https://hf-mirror.com

#      export HF_HUB_ETAG_TIMEOUT=120
